ac_convert_to_seven = {
	picture = "gfx\interface\illustrations\decisions\decision_major_religion.dds"

	major = yes

	desc = ac_convert_to_seven_desc
	selection_tooltip = ac_convert_to_seven_tt

	ai_check_interval = 12

	is_shown = {
		faith = faith:valyrian
		root = character:Targaryen_27
		NOT = {
			has_title = title:e_the_iron_throne
		}
		is_ai = no
	}

	is_valid = {
		faith = faith:valyrian
	}

	effect = {
		set_character_faith = faith:the_seven_main
		
		character:Targaryen_28 ?= { # Visenya
			if = {
				limit = { is_alive = yes }
				set_character_faith = faith:the_seven_main
			}
		}
		character:Targaryen_29 ?= { # Aegon
			if = {
				limit = { is_alive = yes }
				set_character_faith = faith:the_seven_main
			}
		}
		title:c_dragonstone = {
			set_county_faith = faith:the_seven_main
		}
		hidden_effect = {
			divorce = character:Targaryen_29 # Rhaenys Targaryen
			agot_safe_concubine_history_effect = {
				TARGET = character:Targaryen_29 # Rhaenys Targaryen
			}
		}
	}

	ai_potential = {
		always = yes
	}

	ai_will_do = {
		base = 100
	}
}
